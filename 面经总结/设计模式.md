##### 简单工厂（静态方法工厂）

****

```javascript
// 创建单一对象的实例
class Factory{
    // 需要创建实例的对象
    constructor(username, pwd, role){
       	this.username = username
        this.pwd = pwd
        this.role = role
    }
}

class CreateRoleFactory{
    // 创建实例的工厂
    static create(username, pwd, role){
        return new Factory(username, pwd, role)
    }
}
let admin = CreateRoleFactory.create('张三', '123456', 'admin')
```

#### 发布-订阅模式

```javascript
class EventBus{
  constructor(){
    this.events = {}
    this.onceEvents = {}
  }
  $on(event, cb){
    if(!this.events[event]){
      this.events[event] = []
    }
    this.events[event].push(cb)
    return this
  }
  $once(event, cb){
    if(!this.onceEvents[event]){
      this.onceEvents[event] = []
    }
    this.onceEvents[event].push(cb)
    return this
  }
  $off(event, cb){
    if(arguments.length === 0){
      this.events = {}
      this.onceEvents = {}
    }else if(arguments.length === 1 &&
      typeof event === 'string'){
      delete this.events[event]
      delete this.onceEvents[event]
    }else if(arguments.length === 2 &&
    typeof event === 'string' &&
    typeof cb === 'function'){
      this.events[event] = this.events[event].filter(x => x !== cb)
      this.onceEvents[event] = this.onceEvents[event].filter(x => x !== cb)
    }else{
      throw new Error('args is wrong')
    }
      return this
  }
  $emit(event, ...args){
    this.events[event].forEach(cb => {
      cb.apply(this, args)
    })
    this.onceEvents[event].forEach(cb => {
      cb.apply(this, args)
    })
    delete this.onceEvents[event]
    return this
  }
}
```
