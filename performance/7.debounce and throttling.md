#### 函数防抖（debounce）

> 一般定义：对于 **一直连续触发** 的事件，防抖的含义就是让之前的之前所有处理函数都失效，事件处理函数只执行一次

防抖有 **延迟防抖** 和 **立即防抖** 两个版本

立即防抖就是执行第一次；延迟防抖就是执行最后一次；

还可以加上取消 `debounce 函数` 的功能

```javascript
function debounce(func, delay, immediate){
    let timeout, result, debounced
    debounced = function(){
        if(timeout){
            clearTimeout(timeout)
        }
        if(immediate){
            // 立即执行
            let callNow = !timeout
            timeout = setTimeout(() => {
                // 刷新 timeout
                timeout = null
            }, delay)
            if(callNow){
                // 只有当timeout为假值的时候，才执行动作
                result = func.apply(this, arguments)
            }
        }else{
            // 延迟执行
            timeout = setTimeout(() => {
				func.apply(this, arguments)
			}, delay)
        }
        return result
    }
    debounced.cancel = function(){
        // 取消 debounced
        clearTimeout(timeout)
        timeout = null
    }
    return debounced
}
```



#### 函数节流（throttling）

> 一般定义：对于 **一定时间期限内连续触发** 的事件，节流的含义就是让所有处理函数都失效，事件处理函数只执行一次

```javascript
// 通过时间戳实现
// 当触发事件的时候，我们取出当前的时间戳，然后减去之前的时间戳(最一开始值设为 0 )，如果大于设置的时间周期，就执行函数，然后更新时间戳为当前的时间戳，如果小于，就不执行
function throttling(func, delay){
    let start = new Date().getTime()
    return function(){
        if(new Date().getTime() - start >= delay){
            func.apply(this, arguments)
            start = new Date().getTime()
    	}
    }
}
```

```javascript
// 通过定时器实现
// 当触发事件的时候，我们设置一个定时器，再触发事件的时候，如果定时器存在，就不执行，直到定时器执行，然后执行函数，清空定时器，这样就可以设置下个定时器
function throttling(func, delay){
    let timeout
    return function(){
        if(!timeout){
            timeout = setTimeout(() => {
                timeout = null
                func.apply(this, arguments)
            }, delay)
    	}
    }
}
```

```javascript
// 结合时间戳和定时器
// 开始做一次，结束后再做一次
function throttling(func, delay){
    let timeout,
        start = new Date().getTime()
    return function(){
        if(timeout){
            clearTimeout(timeout)
        }
        if(new Date().getTime() - start >= delay){
            func.apply(this, arguments)
            start = new Date().getTime()
        }else {
            timeout = setTimeout(() => {
                timeout = null
                func.apply(this, arguments)
            }, delay)
    	}
    }
}
```

```javascript
// 通过 options 控制
function throttle(func, delay, options){
  let timeout,
      start = new Date().getTime(),
      throttled
  options = options || {timeStamp: true, timer: true}
  throttled = function(){
    if(new Date().getTime() - start >= delay && options.timeStamp){
      // options.timeStamp === true 的时候，节流器加上开始做一次
      if(timeout){
        clearTimeout(timeout)
        timeout = null
      }
      func.apply(this, arguments)
      start = new Date().getTime()
    }else if(!timeout && options.timer){
      // options.timer === true，节流器加上结束后做一次
      timeout = setTimeout(() => {
        timeout = null
        func.apply(this, arguments)
      }, delay)
    }
  }
  throttled.cancel = function(){
      clearTimeout(timeout)
      start = new Date().getTime()
      timeout = null
  }
}
```
